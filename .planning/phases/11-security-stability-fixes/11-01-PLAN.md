**Implementation approach for large guest tables:** Since Convex mutations have document read/write limits, and a draft event with 60K guests cannot be deleted in a single mutation, use a chunked approach:

1. First, delete the smaller tables (foodRules, foodScans, smsDeliveries, deviceSessions, cardTemplates) -- these are expected to be small for draft events.
2. For guests, use a paginated loop within the mutation. Convex allows reading and deleting up to ~8000 documents per mutation. Since draft events are unlikely to have 60K guests (imports happen during active/live phases), a single mutation with a loop should suffice. Add a safety limit:

```typescript
// Delete guests in chunks (draft events typically have < 1000 guests)
let guestCursor = null;
let totalDeleted = 0;
const MAX_GUEST_DELETES = 8000; // Convex mutation limit safety

do {
  const guestBatch = await ctx.db
    .query("guests")
    .withIndex("by_event", (q) => q.eq("eventId", args.eventId))
    .take(500);

  if (guestBatch.length === 0) break;

  for (const guest of guestBatch) {
    await ctx.db.delete(guest._id);
    totalDeleted++;
  }

  if (totalDeleted >= MAX_GUEST_DELETES) {
    throw new Error(
      `Event has too many guests (${totalDeleted}+) for single-mutation deletion. Contact admin.`
    );
  }
} while (true);
```

1. `foodScans` (references guests and stalls)
2. `smsDeliveries` (references guests)
3. `foodRules` (references guestCategories and vendorCategories)
4. `deviceSessions` (references stalls and vendorTypes)
5. `cardTemplates` (references events only)
6. `guests` (large table -- chunked)
7. `guestCategories` (existing)
8. `stalls` (existing)
9. `vendorCategories` (existing)
10. `vendorTypes` (existing)
11. `events` (the event itself -- last)

Each table query uses the `by_event` index with `.collect()` (small tables) or `.take(500)` loop (guests).

**1. Create `frontend/src/components/error-boundary.tsx`:**

Build a class component (ErrorBoundary must be a class component in React):

```tsx
import { Component, type ErrorInfo, type ReactNode } from "react";

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error: Error | null;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error("[ErrorBoundary]", error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      if (this.props.fallback) return this.props.fallback;
      return <ErrorFallback error={this.state.error} onRetry={() => this.setState({ hasError: false, error: null })} />;
    }
    return this.props.children;
  }
}
```

```typescript
const recipients: Array<{guestId: Id<"guests">; name: string; phone: string; cardUrl: string}> = [];
let isDone = false;
let cursor: string | null = null;

while (!isDone) {
  const page = await ctx.db
    .query("guests")
    .withIndex("by_event", (q) => q.eq("eventId", args.eventId))
    .paginate({ numItems: 500, cursor: cursor ?? undefined });

        guestId: guest._id,
        name: guest.name,
```

```typescript
/**
 * Signs a payload with HMAC-SHA256 using the Web Crypto API.
 * Returns the hex-encoded signature string.
 */
export async function signPayload(
  secret: string,
  timestamp: string,
  body: string,
): Promise<string> {
  const encoder = new TextEncoder();
  const keyData = encoder.encode(secret);
  const key = await crypto.subtle.importKey(
    "raw",
    keyData,
    { name: "HMAC", hash: "SHA-256" },
    false,
    ["sign"],
  );

  const signatureBuffer = await crypto.subtle.sign(
    "HMAC",
    key,
    encoder.encode(timestamp + body),
  );

  return Array.from(new Uint8Array(signatureBuffer))
    .map((b) => b.toString(16).padStart(2, "0"))
    .join("");
}

/**
 * Constant-time string comparison for HMAC signatures.
 * Uses crypto.subtle.timingSafeEqual if available (Node 16+),
 * falls back to byte-level XOR comparison.
 */
export function timingSafeEqual(a: string, b: string): boolean {
  if (a.length !== b.length) return false;

  const encoder = new TextEncoder();
  const bufA = encoder.encode(a);
  const bufB = encoder.encode(b);

  let result = 0;
  for (let i = 0; i < bufA.length; i++) {
    result |= bufA[i] ^ bufB[i];
  }
  return result === 0;
}
```
