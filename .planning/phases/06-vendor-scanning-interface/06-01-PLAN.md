    **frontend/src/hooks/use-device-session.ts:**
    Create a hook managing device session lifecycle:
    ```typescript
    const SESSION_KEY = 'eventarc_scanner_session';
    
    interface SessionInfo {
      token: string;
      stallId: string;
      eventId: string;
      vendorCategoryId: string;
      vendorTypeId: string;
      stallName: string;
    }
    
    export function useDeviceSession() {
      const [token, setToken] = useState<string | null>(null);
      const [session, setSession] = useState<SessionInfo | null>(null);
      const [isLoading, setIsLoading] = useState(true);
      const [isRevoked, setIsRevoked] = useState(false);
      
      // On mount: check localStorage for existing token, validate with backend
      useEffect(() => {
        const stored = localStorage.getItem(SESSION_KEY);
        if (!stored) { setIsLoading(false); return; }
        // Validate token with GET /api/v1/session
        validateToken(stored).then(info => {
          if (info) { setToken(stored); setSession(info); }
          else { localStorage.removeItem(SESSION_KEY); }
          setIsLoading(false);
        });
      }, []);
      
      async function createSession(params: {
        stallId: string; eventId: string; vendorCategoryId: string; 
        vendorTypeId: string; stallName: string;
      }): Promise<boolean> {
        const API_URL = import.meta.env.VITE_API_URL || '';
        const res = await fetch(`${API_URL}/api/v1/session`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(params),
        });
        if (!res.ok) return false;
        const { token: newToken } = await res.json();
        localStorage.setItem(SESSION_KEY, newToken);
        setToken(newToken);
        setSession({ token: newToken, ...params });
        return true;
      }
      
      function clearSession() {
        localStorage.removeItem(SESSION_KEY);
        setToken(null);
        setSession(null);
      }
      
      return { token, session, isLoading, isRevoked, createSession, clearSession };
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Vendor device -> Go API | Untrusted device creates session, validates session. No authentication -- any device can create a session. |
| Admin dashboard -> Go API | Authenticated admin revokes sessions via HMAC-protected endpoint. |
| Go API -> Redis | Session tokens stored in Redis. Internal boundary (trusted). |
| Go API -> Convex | Session metadata synced to Convex for admin monitoring. Internal boundary (trusted). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-01 | Spoofing | POST /api/v1/session | accept | Session tokens grant stall-scoped access only, no PII. Brute-force infeasible (256-bit token space). Rate limiting can be added if needed. |
| T-6-02 | Spoofing | GET /api/v1/session | mitigate | Token validated against Redis on every request. Invalid/revoked tokens rejected with 401. |
| T-6-03 | Tampering | DELETE /api/v1/admin/session | mitigate | Revocation endpoint is HMAC-protected (admin-only). Vendor devices cannot revoke sessions. |
| T-6-04 | Information Disclosure | GET /api/v1/session response | mitigate | Response contains only stallId, eventId, vendorCategoryId, vendorTypeId, createdAt. No guest PII, no admin data. |
| T-6-05 | Denial of Service | POST /api/v1/session | accept | Session creation is lightweight (Redis SET). Flooding would fill Redis but tokens are small. Add rate limiting per IP if abuse observed. |
| T-6-06 | Elevation of Privilege | localStorage token theft | accept | Stolen token only allows scanning at assigned stall -- no admin access, no PII exposure. Admin can revoke. |
</threat_model>
