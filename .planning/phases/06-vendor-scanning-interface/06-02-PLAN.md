    **frontend/src/hooks/use-audio-feedback.ts:**
    Create hook that manages AudioContext lifecycle:
    ```typescript
    export function useAudioFeedback() {
      const audioCtxRef = useRef<AudioContext | null>(null);
      
      // Initialize AudioContext on first user interaction
      function ensureContext() {
        if (!audioCtxRef.current) {
          audioCtxRef.current = new AudioContext();
        }
        if (audioCtxRef.current.state === 'suspended') {
          audioCtxRef.current.resume();
        }
        return audioCtxRef.current;
      }
      
      function playTone(frequency: number, duration: number) {
        const ctx = ensureContext();
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();
        osc.connect(gain);
        gain.connect(ctx.destination);
        osc.frequency.value = frequency;
        osc.type = 'sine';
        gain.gain.value = 0.3;
        osc.start();
        osc.stop(ctx.currentTime + duration / 1000);
      }
      
      function play(type: AudioCueType) {
        const cue = AUDIO_CUES[type];
        cue.tones.forEach((tone, i) => {
          setTimeout(() => playTone(tone.freq, tone.dur), i * (cue.delay + cue.tones[0].dur));
        });
      }
      
      return { play, playSuccess: () => play('success'), playFailure: () => play('failure'), playDuplicate: () => play('duplicate') };
    }
    ```

    **frontend/src/hooks/use-scanner.ts:**
    Create Zustand store for scan state machine:
    ```typescript
    type ScanState = 'idle' | 'scanned' | 'flash' | 'reviewing' | 'confirmed' | 'dismissed' | 'ready';
    
    interface ScanResult {
      qrPayload: string;
      decodedAt: number;
    }
    
    type ScanOutcome = 'allowed' | 'served' | 'denied' | 'duplicate_entry' | 'duplicate_food';
    
    interface ServerResponse {
      outcome: ScanOutcome;
      guestName?: string;
      guestCategory?: string;
      foodCategory?: string;
      used?: number;
      limit?: number;
      remaining?: number;
      reason?: string;
      originalCheckIn?: { time: string; stall: string };
      consumptionHistory?: Array<{ stall: string; time: string }>;
    }
    
    interface ScanStore {
      state: ScanState;
      scanResult: ScanResult | null;
      serverResponse: ServerResponse | null;
      scanCount: number;
      
      onQrDetected: (qrPayload: string) => void;
      onFlashComplete: () => void;
      onConfirm: (sessionToken: string, stallId: string, vendorTypeId: string) => Promise<void>;
      onDismiss: () => void;
      onScanNext: () => void;
      reset: () => void;
    }
    ```
    
    State transitions:
    - `onQrDetected`: Only from 'idle'. Set scanResult, transition to 'scanned', then immediately to 'flash'. Ignore if not 'idle' (prevents double-scan per Pitfall 4).
    - `onFlashComplete`: From 'flash' to 'reviewing'. Called after 1-second flash timer.
    - `onConfirm`: From 'reviewing'. Sends POST to /api/v1/scan/entry or /api/v1/scan/food (based on vendorTypeId -- "entry" type uses entry endpoint, "food" type uses food endpoint with stallId). Parse response into ServerResponse. Transition to 'confirmed', then to 'ready'.
    - `onDismiss`: From 'reviewing' to 'dismissed', then to 'ready'. No server call (D-02: dismiss cancels).
    - `onScanNext`: From 'ready' to 'idle'. Increment scanCount.
    
    The onConfirm function determines endpoint by checking vendorTypeId:
    - If vendorType is "entry": POST /api/v1/scan/entry with { qrPayload }
    - If vendorType is "food": POST /api/v1/scan/food with { qrPayload, stallId }
    
    Include the session token as Bearer header for scan requests.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-07 | Tampering | QR payload from camera | mitigate | QR payload validated server-side via HMAC signature (Phase 4). Scanner sends raw decoded text -- server validates. |
| T-6-08 | Information Disclosure | Camera feed | mitigate | Camera stream stays within browser sandbox. html5-qrcode does not transmit video -- only decoded QR text is sent to server. Stream released on component unmount. |
| T-6-09 | Denial of Service | Rapid scan flooding | accept | Two-step confirm flow (D-02) and manual "Scan Next" (D-03) naturally rate-limit scans to human speed (~1 scan per 3-5 seconds). Server-side HMAC validation rejects invalid payloads cheaply. |
| T-6-10 | Spoofing | Admin session revocation | mitigate | Revoke mutation called via Convex (authenticated via Better Auth). Go admin revoke endpoint is HMAC-protected. |
| T-6-11 | Repudiation | Scan actions | mitigate | Every confirmed scan creates a server-side record (Phase 4/5). Dismissed scans have no server record (by design -- D-02). |
