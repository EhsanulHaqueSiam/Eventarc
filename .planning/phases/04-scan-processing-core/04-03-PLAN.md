```javascript
import http from 'k6/http';
import { check, sleep } from 'k6';
import { SharedArray } from 'k6/data';
import { Counter, Rate, Trend } from 'k6/metrics';

// Custom metrics
const scanSuccess = new Counter('scan_success');
const scanDuplicate = new Counter('scan_duplicate');
const scanError = new Counter('scan_error');
const scanDuration = new Trend('scan_duration');

// Pre-generated QR payloads (one per VU, loaded from JSON file)
const payloads = new SharedArray('payloads', function() {
  return JSON.parse(open('./payloads.json'));
});

export const options = {
  scenarios: {
    // Scenario 1: Ramp up to 10K concurrent VUs
    concurrent_scans: {
      executor: 'ramping-vus',
      startVUs: 0,
      stages: [
        { duration: '10s', target: 1000 },   // Warm up
        { duration: '10s', target: 5000 },   // Ramp to 5K
        { duration: '10s', target: 10000 },  // Full 10K
        { duration: '30s', target: 10000 },  // Sustain 10K for 30s
        { duration: '10s', target: 0 },      // Ramp down
      ],
    },
    // Scenario 2: Duplicate scan flood (same QR scanned repeatedly)
    duplicate_flood: {
      executor: 'constant-vus',
      vus: 100,
      duration: '20s',
      startTime: '70s', // Start after main scenario
    },
  },
  thresholds: {
    http_req_failed: ['rate==0'],             // Zero HTTP errors
    http_req_duration: ['p(95)<200'],         // p95 < 200ms
    scan_error: ['count==0'],                 // Zero scan processing errors
  },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:8080';

export default function() {
  const idx = __VU - 1;  // VU index (0-based)
  const payload = payloads[idx % payloads.length];
  
  const res = http.post(`${BASE_URL}/api/v1/scan/entry`, JSON.stringify({
    qr_payload: payload.qr_payload,
    stall_id: payload.stall_id || 'stall_load_test',
    device_id: `device_k6_${__VU}`,
  }), {
    headers: { 'Content-Type': 'application/json' },
    timeout: '5s',
  });

  scanDuration.add(res.timings.duration);

  check(res, {
    'status is 200 or 409': (r) => r.status === 200 || r.status === 409,
    'response has status field': (r) => {
      const body = JSON.parse(r.body);
      return body.status === 'valid' || body.status === 'duplicate';
    },
  });

  if (res.status === 200) scanSuccess.add(1);
  else if (res.status === 409) scanDuplicate.add(1);
  else scanError.add(1);
}
```

```go
package scan

// TestConcurrentEntryScan_NoDuplicates verifies that N concurrent scans for N different guests
// produce exactly N successful check-ins and zero duplicates.
//
// Setup: miniredis + 500 seeded guests
// Execute: 500 goroutines, each scanning one unique guest simultaneously
// Verify:
//   - Exactly 500 scans return "valid"
//   - Zero scans return error
//   - Redis SCARD checkedin:{eventId} == 500
//   - Redis HGET counters:{eventId} attendance == "500"
func TestConcurrentEntryScan_NoDuplicates(t *testing.T)

// TestConcurrentDuplicateScan_OnlyOneSucceeds verifies that N concurrent scans for the SAME guest
// produce exactly 1 successful check-in and N-1 "duplicate" responses.
//
// Setup: miniredis + 1 seeded guest
// Execute: 100 goroutines, all scanning the same guest simultaneously
// Verify:
//   - Exactly 1 scan returns "valid"
//   - Exactly 99 scans return "duplicate"
//   - Zero errors
//   - Redis SCARD checkedin:{eventId} == 1
//   - Redis HGET counters:{eventId} attendance == "1"
func TestConcurrentDuplicateScan_OnlyOneSucceeds(t *testing.T)

// TestConcurrentMixedScan_CorrectCounts verifies a mixed workload of unique and duplicate scans.
//
// Setup: miniredis + 200 guests, each scanned 5 times concurrently (1000 total goroutines)
// Verify:
//   - Exactly 200 "valid" results
//   - Exactly 800 "duplicate" results
//   - Zero errors
//   - Redis attendance counter == 200
func TestConcurrentMixedScan_CorrectCounts(t *testing.T)

// TestConcurrentCounterAccuracy verifies atomic counter accuracy under concurrent load.
//
// Setup: miniredis + 3 categories (vip: 50, regular: 100, staff: 50)
// Execute: 200 goroutines scanning simultaneously
// Verify:
//   - counters:{eventId} attendance == "200"
//   - counters:{eventId} vip:checkedin == "50"
//   - counters:{eventId} regular:checkedin == "100"
//   - counters:{eventId} staff:checkedin == "50"
func TestConcurrentCounterAccuracy(t *testing.T)
```

Each test must:
1. Create miniredis instance
2. Seed test data
3. Use `sync.WaitGroup` to synchronize goroutine starts
4. Use `sync.Barrier` (or channel-based barrier) to ensure all goroutines start simultaneously
5. Collect results in thread-safe manner (`sync.Mutex`-protected slice)
6. Assert exact counts after all goroutines complete
7. Verify Redis state matches expected counts

These tests MUST pass with `go test -race ./internal/scan/... -run TestConcurrent` — the race detector validates no shared state data races.

```bash
#!/usr/bin/env bash
set -euo pipefail

# Load Test Runner for Phase 4: Scan Processing Core
# Prerequisites: Docker, k6, Go
# Usage: ./backend/tests/load/run.sh [--count 15000] [--vus 10000] [--duration 30s]

COUNT=${1:-15000}
EVENT_ID="load_test_$(date +%s)"
HMAC_SECRET="load_test_hmac_secret"
BASE_URL="http://localhost:8080"

echo "=== Phase 4 Load Test ==="
echo "Guests: $COUNT"
echo "Event: $EVENT_ID"

# Step 1: Ensure infrastructure is running
echo ">>> Starting infrastructure..."
docker compose up -d postgres pgbouncer redis
sleep 3

# Step 2: Run migrations
echo ">>> Running migrations..."
# golang-migrate command or manual psql apply

# Step 3: Start server (background)
echo ">>> Starting server..."
cd backend && HMAC_SECRET=$HMAC_SECRET go run ./cmd/server &
SERVER_PID=$!
sleep 2

# Step 4: Seed test data
echo ">>> Seeding $COUNT guests..."
cd backend && go run ./tests/load/seed.go -count $COUNT -event $EVENT_ID -hmac-secret $HMAC_SECRET

# Step 5: Start worker (background)
echo ">>> Starting worker..."
cd backend && go run ./cmd/worker &
WORKER_PID=$!
sleep 1

# Step 6: Run k6 load test
echo ">>> Running k6 load test..."
k6 run --env BASE_URL=$BASE_URL backend/tests/load/scan_load_test.js

# Step 7: Verify counter reconciliation
echo ">>> Verifying counter reconciliation..."
REDIS_ATTENDANCE=$(redis-cli HGET "counters:$EVENT_ID" attendance)
PG_COUNT=$(psql -t -c "SELECT COUNT(*) FROM entry_scans WHERE event_id='$EVENT_ID' AND status='valid'" "$DATABASE_URL")
echo "Redis attendance: $REDIS_ATTENDANCE"
echo "PG scan count: $PG_COUNT"

if [ "$REDIS_ATTENDANCE" = "$PG_COUNT" ]; then
  echo ">>> PASS: Counters match"
else
  echo ">>> FAIL: Counter mismatch (Redis=$REDIS_ATTENDANCE, PG=$PG_COUNT)"
  exit 1
fi

# Cleanup
kill $SERVER_PID $WORKER_PID 2>/dev/null || true
echo "=== Load Test Complete ==="
```

Make the script executable: `chmod +x backend/tests/load/run.sh`
