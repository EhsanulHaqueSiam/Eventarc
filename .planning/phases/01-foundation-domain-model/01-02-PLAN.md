```typescript
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  events: defineTable({
    name: v.string(),
    description: v.optional(v.string()),
    venue: v.optional(v.string()),
    eventDate: v.number(),
    endDate: v.optional(v.number()),
    status: v.union(
      v.literal("draft"),
      v.literal("active"),
      v.literal("live"),
      v.literal("completed"),
      v.literal("archived"),
    ),
    config: v.object({
      qrStrategy: v.union(v.literal("unified"), v.literal("separate")),
      foodQrMode: v.union(v.literal("guestLinked"), v.literal("anonymous")),
      foodQrTiming: v.union(v.literal("preSent"), v.literal("postEntry")),
    }),
    createdBy: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_status", ["status"])
    .index("by_createdBy", ["createdBy"]),

  guestCategories: defineTable({
    eventId: v.id("events"),
    name: v.string(),
    isDefault: v.boolean(),
  })
    .index("by_event", ["eventId"])
    .index("by_event_name", ["eventId", "name"]),
});
```

Note: The schema uses camelCase per D-26 Convex convention. `eventDate` is a Unix timestamp (number). `config` is a nested object per D-19 research recommendation (no separate table needed).

**1. Create `convex/model/events.ts`** -- Business logic for event lifecycle (per D-07):

```typescript
import type { MutationCtx, QueryCtx } from "../_generated/server";
import type { Id } from "../_generated/dataModel";

// State machine transitions (D-07)
// draft -> active: Configuration complete, ready for guest import/QR
// active -> live: Event goes live for scanning (auto or manual)
// active -> draft: Revert to draft for changes (only before going live)
// live -> completed: Event ends
// completed -> archived: Long-term storage
const VALID_TRANSITIONS: Record<string, string[]> = {
  draft: ["active"],
  active: ["live", "draft"],
  live: ["completed"],
  completed: ["archived"],
  archived: [],
};

export function getValidTransitions(currentStatus: string): string[] {
  return VALID_TRANSITIONS[currentStatus] ?? [];
}

export function validateTransition(
  currentStatus: string,
  newStatus: string,
): void {
  const allowed = VALID_TRANSITIONS[currentStatus];
  if (!allowed || !allowed.includes(newStatus)) {
    throw new Error(
      `Invalid transition: ${currentStatus} -> ${newStatus}. ` +
      `Allowed: ${allowed?.join(", ") ?? "none"}`
    );
  }
}

export function isConfigLocked(status: string): boolean {
  // Config is locked once event goes live -- scanning may have started
  return status === "live" || status === "completed" || status === "archived";
}
```

Export `VALID_TRANSITIONS`, `getValidTransitions`, `validateTransition`, and `isConfigLocked`.

**2. Create `convex/model/categories.ts`** -- Category business logic (per D-10, D-11):

```typescript
import type { MutationCtx } from "../_generated/server";
import type { Id } from "../_generated/dataModel";

export async function createDefaultCategory(
  ctx: MutationCtx,
  eventId: Id<"events">,
): Promise<Id<"guestCategories">> {
  return await ctx.db.insert("guestCategories", {
    eventId,
    name: "General",
    isDefault: true,
  });
}

export async function validateCategoryName(
  ctx: MutationCtx,
  eventId: Id<"events">,
  name: string,
  excludeId?: Id<"guestCategories">,
): Promise<void> {
  const existing = await ctx.db
    .query("guestCategories")
    .withIndex("by_event_name", (q) => q.eq("eventId", eventId).eq("name", name))
    .first();
  if (existing && existing._id !== excludeId) {
    throw new Error(`Category "${name}" already exists for this event`);
  }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> Convex | Unauthenticated users must not mutate event/category data |
| Convex mutation auth | Every public mutation must check ctx.auth.getUserIdentity() |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | Convex mutations | mitigate | Every public mutation calls `ctx.auth.getUserIdentity()` and throws if null. Better Auth manages session tokens. |
| T-02-02 | Tampering | Event config after live | mitigate | `isConfigLocked()` check in `updateConfig` mutation prevents config changes after event goes live. |
| T-02-03 | Tampering | Event state transitions | mitigate | `validateTransition()` enforces the state machine. Cannot skip states or go backwards from live. |
| T-02-04 | Denial of Service | Event deletion cascade | accept | Cascade delete of categories/vendors on event delete could be slow for events with many children. Acceptable because deletion only allowed for draft events (no children yet). |
| T-02-05 | Information Disclosure | Convex queries | mitigate | Convex argument validators (v.string, v.id, v.union) reject malformed input at the API boundary. No raw user input reaches the database. |
| T-02-06 | Elevation of Privilege | Admin-only mutations | mitigate | Auth identity check on every mutation. Vendor device sessions (Phase 6) will have separate, limited access. |
</threat_model>
