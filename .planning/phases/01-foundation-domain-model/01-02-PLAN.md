import { createClient, type GenericCtx } from "@convex-dev/better-auth";
import { convex, crossDomain } from "@convex-dev/better-auth/plugins";
import { components } from "./_generated/api";
import { DataModel } from "./_generated/dataModel";
import { query } from "./_generated/server";
import { betterAuth } from "better-auth/minimal";
import authConfig from "./auth.config";

const siteUrl = process.env.SITE_URL ?? "http://localhost:5173";

export const authComponent = createClient<DataModel>(components.betterAuth);

export const createAuth = (ctx: GenericCtx<DataModel>) => {
  return betterAuth({
    trustedOrigins: [siteUrl],
    database: authComponent.adapter(ctx),
    emailAndPassword: {
      enabled: true,
      requireEmailVerification: false,
    },
    plugins: [
      crossDomain({ siteUrl }),
      convex({ authConfig }),
    ],
  });
};

export const getCurrentUser = query({
  args: {},
  handler: async (ctx) => {
    return authComponent.getAuthUser(ctx);
  },
});
```

import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  events: defineTable({
    name: v.string(),
    description: v.optional(v.string()),
    venue: v.optional(v.string()),
    eventDate: v.number(),       // Unix timestamp in milliseconds
    endDate: v.optional(v.number()),
    status: v.union(
      v.literal("draft"),
      v.literal("active"),
      v.literal("live"),
      v.literal("completed"),
      v.literal("archived"),
    ),
    config: v.object({
      qrStrategy: v.union(v.literal("unified"), v.literal("separate")),
      foodQrMode: v.union(v.literal("guestLinked"), v.literal("anonymous")),
      foodQrTiming: v.union(v.literal("preSent"), v.literal("postEntry")),
    }),
    createdBy: v.optional(v.string()),
    createdAt: v.number(),
    updatedAt: v.number(),
  })
    .index("by_status", ["status"])
    .index("by_createdBy", ["createdBy"]),

  guestCategories: defineTable({
    eventId: v.id("events"),
    name: v.string(),
    isDefault: v.boolean(),
  })
    .index("by_event", ["eventId"])
    .index("by_event_name", ["eventId", "name"]),

  vendorTypes: defineTable({
    eventId: v.id("events"),
    name: v.union(v.literal("entry"), v.literal("food")),
  })
    .index("by_event", ["eventId"])
    .index("by_event_name", ["eventId", "name"]),

  vendorCategories: defineTable({
    eventId: v.id("events"),
    vendorTypeId: v.id("vendorTypes"),
    name: v.string(),
  })
    .index("by_event", ["eventId"])
    .index("by_vendorType", ["vendorTypeId"])
    .index("by_event_name", ["eventId", "name"]),

  stalls: defineTable({
    eventId: v.id("events"),
    categoryId: v.id("vendorCategories"),
    name: v.string(),
    isActive: v.boolean(),
  })
    .index("by_event", ["eventId"])
    .index("by_category", ["categoryId"])
    .index("by_event_name", ["eventId", "name"]),
});
```

**1. convex/model/events.ts** (business logic helpers):

```typescript
import { MutationCtx, QueryCtx } from "../_generated/server";
import { Id } from "../_generated/dataModel";

// D-07: Valid state transitions
// draft -> active (configuration complete, ready for guests)
// active -> live (scanning starts -- automatic at eventDate or manual "Go Live Early")
// active -> draft (admin reverts to make changes, only before going live)
// live -> completed (event ends)
// completed -> archived (long-term storage)
export const VALID_TRANSITIONS: Record<string, string[]> = {
  draft: ["active"],
  active: ["live", "draft"],
  live: ["completed"],
  completed: ["archived"],
  archived: [],
};

export function validateTransition(
  currentStatus: string,
  newStatus: string,
): void {
  const allowed = VALID_TRANSITIONS[currentStatus];
  if (!allowed || !allowed.includes(newStatus)) {
    throw new Error(
      `Invalid transition: ${currentStatus} -> ${newStatus}. Allowed: ${allowed?.join(", ") || "none"}`
    );
  }
}

// D-11: Auto-create default "General" category
export async function createDefaultCategory(
  ctx: MutationCtx,
  eventId: Id<"events">,
): Promise<Id<"guestCategories">> {
  return await ctx.db.insert("guestCategories", {
    eventId,
    name: "General",
    isDefault: true,
  });
}

// D-08: Auto-create "entry" and "food" vendor types
export async function createDefaultVendorTypes(
  ctx: MutationCtx,
  eventId: Id<"events">,
): Promise<void> {
  await ctx.db.insert("vendorTypes", { eventId, name: "entry" });
  await ctx.db.insert("vendorTypes", { eventId, name: "food" });
}
```

**2. convex/events.ts** (public API -- thin wrappers per D-29):

Exports:
- `create` mutation: Args = {name: string, description?: string, venue?: string, eventDate: number, endDate?: number, config: {qrStrategy, foodQrMode, foodQrTiming}}. Checks auth via `ctx.auth.getUserIdentity()`. Inserts event with status "draft", createdAt/updatedAt = Date.now(), createdBy = identity.subject. Calls `createDefaultCategory(ctx, eventId)` and `createDefaultVendorTypes(ctx, eventId)`. Returns eventId.

- `list` query: Args = {status?: union of status literals}. Returns all events, optionally filtered by status. Ordered by createdAt descending. Does NOT require auth (admin is the only user anyway, but listing is safe).

- `get` query: Args = {eventId: v.id("events")}. Returns the event document or null.

- `updateConfig` mutation: Args = {eventId: v.id("events"), config: {qrStrategy, foodQrMode, foodQrTiming}}. Checks auth. Reads event, verifies status is NOT "live", "completed", or "archived" (config locked after go-live). Updates config and updatedAt.

- `updateDetails` mutation: Args = {eventId: v.id("events"), name?: string, description?: string, venue?: string, eventDate?: number, endDate?: number}. Checks auth. Updates provided fields and updatedAt.

- `updateStatus` mutation: Args = {eventId: v.id("events"), newStatus: union of status literals}. Checks auth. Reads event, calls `validateTransition(event.status, newStatus)`. Updates status and updatedAt. If newStatus is "live", schedule the sync action: `await ctx.scheduler.runAfter(0, api.internal.sync.pushEventToGo, { eventId })` -- BUT since we haven't created the internal sync action yet (Plan 03), just add a TODO comment for now. Log the transition instead.

- `remove` mutation: Args = {eventId: v.id("events")}. Checks auth. Only allowed when status is "draft". Deletes the event AND all related guestCategories, vendorTypes, vendorCategories, and stalls for that event (cascade delete).

**3. convex/model/categories.ts** (business logic helpers):

```typescript
import { MutationCtx } from "../_generated/server";
import { Id } from "../_generated/dataModel";

export async function validateCategoryName(
  ctx: MutationCtx,
  eventId: Id<"events">,
  name: string,
  excludeId?: Id<"guestCategories">,
): Promise<void> {
  const existing = await ctx.db
    .query("guestCategories")
    .withIndex("by_event_name", (q) => q.eq("eventId", eventId).eq("name", name))
    .first();
  if (existing && existing._id !== excludeId) {
    throw new Error(`Category "${name}" already exists for this event`);
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> Convex | All Convex mutations require auth via Better Auth. Unauthenticated requests rejected. |
| Convex schema validation | Convex validates all mutation args against schema validators (v.string, v.union, v.id). Invalid data types rejected at API boundary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | Convex mutations | mitigate | Every public mutation calls `ctx.auth.getUserIdentity()` and throws "Not authenticated" if null. Better Auth handles session tokens. |
| T-02-02 | Tampering | Event state transitions | mitigate | `validateTransition()` enforces allowed transitions. Cannot skip states or go backward (except active->draft). |
| T-02-03 | Tampering | Event config after go-live | mitigate | `updateConfig` mutation checks status is not live/completed/archived. Config locked once scanning starts. |
| T-02-04 | Information Disclosure | Event listing | accept | `list` query does not require auth. Events are admin-only data, and there is only one admin user class. Low risk. |
| T-02-05 | Denial of Service | Category creation | mitigate | `validateCategoryName` prevents duplicate category names per event. Schema validators prevent oversized strings (Convex default limits). |
| T-02-06 | Elevation of Privilege | Default category deletion | mitigate | `remove` mutation explicitly checks `isDefault === true` and throws error. Cannot remove the General category. |
</threat_model>
